    def get_uct(self, parent_visits: int, exploration_constant: float = 1.41) -> float:
        if self.visits == 0:
            return float("inf")
        return (self.value / self.visits) + exploration_constant * np.sqrt(
            np.log(parent_visits) / self.visits
        )
